-- Lanting Stream Flink SQL
-- 实时用户行为漏斗分析
INSERT INTO dws_user_funnel
SELECT
window_start,
window_end,
channel,
COUNT(DISTINCT user_id) AS uv,
SUM(amount) AS gmv
FROM TABLE(TUMBLE(TABLE dwd_user_event,
DESCRIPTOR(event_time),
INTERVAL '1' MINUTE))
WHERE event_type IN ('view', 'click', 'order')
GROUP BY window_start, window_end, channel
